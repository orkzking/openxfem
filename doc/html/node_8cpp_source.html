177 int Node :: computeNumberOfDofs ()
178 // ********************************
179 // Computes the number of degrees of freedom of the receiver.
180 // Need to include the additional Dofs of enriched nodes (XFEM implementation)
181 // NVP 2005
182 {
183  numberOfDofs = this->readInteger("nDofs") ; // real Dofs ( standard ones )
184 
185  // if the receiver is a non-enriched node, stop. Otherwise it has
186  // ( NSD * number_of_enrichmentfunctions ) additional Dofs for each enrichment item.
187 
188  if ( isEnriched == 0 )
189  return numberOfDofs ;
190 
191  size_t enrichedDofs = 0 ;
192  list<EnrichmentItem*>* enrItemList = this->giveEnrItemListOfNode();
193 
194  list<EnrichmentItem*>::iterator iter ;
195  for(iter = enrItemList->begin(); iter != enrItemList->end(); iter++)
196  {
197  vector<EnrichmentFunction*>* enrFnVector = (*iter)->giveEnrFuncVector();
198  enrichedDofs += 2 * enrFnVector->size();
199  }
200 
201  numberOfDofs += enrichedDofs ;
202 
203  return numberOfDofs ;

496 void Node :: resolveLinearDependency(EnrichmentItem *enrItem)
497 // **********************************************************
498 // Using the area inclusion criterion to remove Heaviside enriched nodes
499 // NVP 2005-06-04
500 {
501  // This tolerance is not constant, but depends on the mesh size h !!!
502  // how to implement this?
503  double eps = 0.0001 ;
504 
505  // get the support of this Node from the Domain
506  map<Node*,vector<Element*> > nodeElemMap = this->domain->giveNodalSupports();
507  vector<Element*> mySupport = nodeElemMap[this] ;
508 
509  double A = 0.0 ; // A is the area of the support
510  double A1 = 0.0 ; // the area above the CrackInterior
511  double temp = 0.0 ;
512 
513  vector<Element*>* elemList = enrItem->giveElementsInteractWithMe();
514  vector<Element*> :: iterator iter ;
515  // Loop on elements in nodal support, say e. If e intersects with enrItem, then
516  // compute the area of e above enrItem by e->giveAreaAboveEnrItem() ;
517  for(size_t i = 0 ; i < mySupport.size() ; i++)
518  {
519  A += mySupport[i]->area();
520  iter = find(elemList->begin(),elemList->end(),mySupport[i]);
521  if (iter != elemList->end())
522  {
523  A1 += mySupport[i]->computeAreaAboveEnrItem(enrItem); // area of part of element above the enrItem
524  temp += mySupport[i]->area() ; // area of elements in nodal support which are split by the crack
525  }
526  }
527  GeometryEntity *geoOfEnrItem = enrItem->giveMyGeo() ;
528  Mu::Point *p = this->makePoint() ;
529  int check = geoOfEnrItem->givePositionComparedTo(p);
530 
531  if(check == 1) // node is above the crack
532  A1 += A - temp ;
533  double A2 = A - A1 ; // the area below the Crack
534 
535  // finally the area inclusion criteria is applied
536  if ((A1/A < eps) || (A2/A < eps))
537  {
538  enrichmentItemListOfNode->remove(enrItem); //do not enrich this node with H(x) any more
539  // after being removed enrichment, if enrichmentItemListOfNode does not contain any
540  // enrichment item, then this node is a classical node !!! NVP 2005-07-12
541  if(enrichmentItemListOfNode->size() == 0 )
542  {
543  isEnriched = 0 ;
544  // update the enriched-nodes-list of DOMAIN !!!
545  this->domain->removeNodeFromEnrichedNodesList(this);
546  }
547  }
548 
549  delete p ;
550  /*
551  // ------------------------- DEBUG --------------------------------------------
552  std::cout<< " checking node " << number << " " << std::endl ;
553  std::cout<< " The first ratio : " << A1/A << std::endl;
554  std::cout<< " The second ratio : " << A2/A << std::endl;
555  // ----------------------------------------------------------------------------
556  */
557 }
