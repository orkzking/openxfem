129 std::list<Element*> CrackTip::buildIntegrationDomain1()
130 // ****************************************************
131 // IntegrationDomain is a set of elements that belong to the ball
132 // of radius r = h*sqrt(area_tip_element) centered at the crack tip.
133 // Remark: This integration domain is no longer used !!!
134 {

152 std::list<Element*> CrackTip::buildIntegrationDomain2()
153 // ****************************************************
154 // IntegrationDomain is a set of elements that intersect with the ball
155 // of radius r = h*sqrt(area_tip_element) centered at the crack tip.
156 {
157  // find elements belongs to the circle.
158 
159  vector<Element*> *interactedElements = this->giveElementsInteractWithMe();
160  Element *tipElem = (*interactedElements)[0] ;
161 
162  Mu::Circle *ball = this->makeBall();
163 
164  std::list<Element*> temp = tipElem->conflicts(ball);
165 
166  // after call Element::conflicts(Circle*), remember to reset checked = false for the next time
167  for(size_t i = 0 ; i < domain->giveNumberOfElements() ; i++)
168  domain->giveElement(i+1)->clearChecked();
169 
170  // find out elements intersecting with the perimeter of the circle
171  // just loop over temp to check, so it is efficient.
172  std::list<Element*> ret ;
173  for(std::list<Element*> :: iterator i = temp.begin() ; i != temp.end() ; i++)
174  {
175  if( (*i)->intersects(ball) )
176  ret.push_back(*i);
177  }
178 
179  delete ball ;
180 
181  return ret;
182 }

185 std::valarray<double> CrackTip::computeInteractionIntegral(TimeStep* stepN)
186 // ************************************************************************
187 // compute the interaction integral, see Yau et al., 1980
188 // Steps :
189 // 1- detection of the elements on which we integrate
190 // 2- loop over these elements
191 // 3- loop over Gauss points
192 // 4- computation of stress, strain... in local coordinates !!! ATTENTION
193 // 5- computation of the auxilliary fields: AuxStress and AuxEps and AuxGradDisp
194 // 6- computation of I1 and I2
195 {
196  FloatArray *sigma,*epsilon,*U ;
197  FloatMatrix *dNdx,*D,*B,*rdu1dx,*rt,*LocalGradDisp ;
198  Node *aNode ;
199  Mu::Point *point,*coord,*globalCoordGP ;
200  GaussPoint *gp;
201  GaussPoint** gaussPointArray;
202  IntArray *nodeArr ;
203  double dV ;
204  size_t numOfNodes ;
205  double dqdx1, dqdx2 ;
206  std::valarray<double> I(0.,2) ; // interaction integral I(1,2) of modes I and II
207  Mu::Circle *myBall = this->makeBall(); // circle centered at tip, r = r_d
208 
209  // Choose domain used for standard XFEM and modified XFEM (crack nucleation)
210  std::list<Element*> intDomain = this->buildIntegrationDomain2();
211 
212  /*
213  // ******************************************************************************
214  // *** EXPORT TO MATLAB FOR PLOT ***
215  // ******************************************************************************
216 
217  // plot the elements in domain A to check.
218  FILE *mlbFile;
219  char mlbFileName[64] = "integration" ; // name of the file
220  strcat(mlbFileName,".m"); // extension .m ( Matlab M file)
221  mlbFile = fopen(mlbFileName, "w");
222 
223  fprintf (mlbFile,"aroundElements = [ ");
224  for(std::list<Element*> :: iterator i = intDomain.begin() ; i != intDomain.end() ; i++)
225  {
226  fprintf (mlbFile,"%d \n",(*i)->giveNumber());
227  }
228  fprintf (mlbFile," ];\n ");
229 
230  // weight functions at nodes ...
231  fprintf (mlbFile,"weight = [ ");
232  double a = myAssociatedCrackInterior->giveCrackLength();
233  std::cout << " Half crack length " << a << endl;
234  for(std::list<Element*> :: iterator i = intDomain.begin() ; i != intDomain.end() ; i++)
235  {
236  std::valarray<double> q(0.,(*i)->giveNumberOfNodes()) ; // weight function q
237  if(this->domain->giveNumberOfInteractedElements()!=0)
238  {
239  for(size_t m = 0 ; m < (*i)->giveNumberOfNodes() ; m++)
240  {
241  aNode = (*i)->giveNode(m+1);
242  point = aNode->makePoint();
243  if (myBall->in(point)) // p locates inside the ball
244  q[m] = 1;
245  delete point ;
246  }
247  }
248  else
249  {
250  double X = this->giveTip()->giveCoordinate(1);
251  double Y = this->giveTip()->giveCoordinate(2);
252  Mu::Point *tip = new Mu::Point(X,Y);
253  for(size_t m = 0 ; m < (*i)->giveNumberOfNodes() ; m++)
254  {
255  aNode = (*i)->giveNode(m+1);
256  point = aNode->makePoint();
257  double r = sqrt(dist(*point,*tip));
258 
259  if(r<=0.25*a)
260  q[m] = 1;
261  else if(r<=a && r>0.25*a)
262  q[m] = 4./3.*(1-r/a) ;
263  else
264  q[m] = 0;
265 
266  delete point ;
267  }
268  }
269  // export to Matlab
270  if( (*i)->giveNumberOfNodes() == 3 )
271  fprintf (mlbFile,"%5.1e %5.1e %5.1e \n",q[0],q[1],q[2]);
272  else if( (*i)->giveNumberOfNodes() == 4 )
273  fprintf (mlbFile,"%5.1e %5.1e %5.1e %5.1e \n",q[0],q[1],q[2],q[3]);
274  else
275  std::cout << " reserve anyone who will do other element types !!! " << std::endl ;
276  }
277  fprintf (mlbFile," ];\n ");
278 
279  fprintf (mlbFile," GP = [ "); // Gauss points for J integral
280 
281  // --------------------------------------------------------------------------------
282  */
283 
284  // inclination angle and rotation matrix
285 
286  FloatMatrix *r = new FloatMatrix(2,2);
287  r->at(1,1) = cos(alpha) ; r->at(1,2) = sin(alpha) ;
288  r->at(2,1) = -sin(alpha) ; r->at(2,2) = cos(alpha) ;
289 
290  rt = r -> GiveTransposition() ;
291 
292  // initialization of used auxiliary field
293  AuxiliaryField<Material,NullMaterial,PlaneStrain> *auxFieldHomo
294  = new AuxiliaryField<Material,NullMaterial,PlaneStrain>();
295 
296  AuxiliaryField<Material,Material,PlaneStrain> *auxFieldBiMat
297  = new AuxiliaryField<Material,Material,PlaneStrain>();
298 
299  // ******************************************************************************
300  // *** LOOP ON ELEMENTS IN THE INTEGRATION DOMAIN ***
301  // ******************************************************************************
302 
303  for(std::list<Element*> :: iterator i = intDomain.begin() ; i != intDomain.end() ; i++)
304  {
305  D = (*i)->giveConstitutiveMatrix()->GiveCopy(); // compliance matrix
306  U = (*i)->ComputeVectorOf('d',stepN); // displacement vector
307  nodeArr = (*i)->giveNodeArray() ; // connectivity of element elem
308  gaussPointArray = (*i)->setGaussQuadForJ_Integral(); // gauss points of elem
309  numOfNodes = (*i)->giveNumberOfNodes() ; // number of nodes of elem
310 
311  // computation of weight function q
312  // Attention : need to rewrite for quadratic XFEM !!!
313  std::valarray<double> q(0.,numOfNodes) ; // weight function q
314 
315  for(size_t m = 0 ; m < numOfNodes ; m++)
316  {
317  aNode = (*i)->giveNode(m+1);
318  point = aNode->makePoint();
319  if (myBall->in(point)) // p locates inside the ball
320  q[m] = 1;
321  delete point ;
322  }
323  /*else
324  {
325  double X = this->giveTip()->giveCoordinate(1);
326  double Y = this->giveTip()->giveCoordinate(2);
327  Mu::Point *tip = new Mu::Point(X,Y);
328  for(size_t m = 0 ; m < numOfNodes ; m++)
329  {
330  aNode = (*i)->giveNode(m+1);
331  point = aNode->makePoint();
332  double r = sqrt(dist(*point,*tip));
333 
334  if(r<=0.25*a)
335  q[m] = 1;
336  else if(r<=a && r>0.25*a)
337  q[m] = 4./3.*(1-r/a) ;
338  else
339  q[m] = 0;
340 
341  delete point ;
342  }
343  }*/
344 
345  // ******************************************************************************
346  // *** LOOP ON GAUSS POINTS OF EACH ELEMENT IN THE INTEGRATION DOMAIN ***
347  // ******************************************************************************
348 
349  for(size_t p = 0 ; p < (*i)->giveNumOfGPtsForJ_Integral() ; p++)
350  {
351  gp = gaussPointArray[p];
352  coord = gp -> giveCoordinates() ;
353  dNdx = (*i)->giveFEInterpolation()->evaldNdx(domain,nodeArr,coord);
354  dV = (*i)->computeVolumeAround(gp);
355 
356  // COMPUTATION OF DERIVATIVES OF U W.R.T x, dU1/dx
357  B = (*i)->ComputeBmatrixAt(gp);
358  FloatMatrix *dU1dx = new FloatMatrix(2,2);// stores derivative of u1 w.r.t x,y
359  for(size_t k = 1 ; k <= 0.5*B->giveNumberOfColumns() ; k++)
360  {
361  dU1dx->at(1,1) += B->at(1,2*k-1)*(*U)[2*k-2]; // du1/dx
362  dU1dx->at(1,2) += B->at(2,2*k) *(*U)[2*k-2]; // du1/dy
363  dU1dx->at(2,1) += B->at(1,2*k-1)*(*U)[2*k-1]; // du2/dx
364  dU1dx->at(2,2) += B->at(2,2*k) *(*U)[2*k-1]; // du2/dy
365  }
366 
367  // COMPUTE DERIVATIVES OF WEIGHT FUNCTION Q
368 
369  //if( this->domain->giveNumberOfInteractedElements() != 0 )
370  //{
371  double dqdx = 0.0 ; double dqdy = 0.0 ;
372  for(size_t m = 0 ; m < numOfNodes ; m++)
373  {
374  dqdx += dNdx->at(m+1,1)*q[m]; // derivative of q w.r.t global x
375  dqdy += dNdx->at(m+1,2)*q[m]; // derivative of q w.r.t global y
376  }
377  dqdx1 = dqdx * cos(alpha) + dqdy * sin(alpha) ;
378  dqdx2 = -dqdx * sin(alpha) + dqdy * cos(alpha) ;
379  //}
380  /*else // compute modified weight function(crack nucleation)
381  {
382  globalCoordGP = (*i)->giveFEInterpolation()->local2Global(domain,nodeArr,coord);
383  FloatArray * localCoord = this->computeLocalCoordOf(globalCoordGP) ;
384 
385  double xloc = (*localCoord)[0];
386  double yloc = (*localCoord)[1];
387  double R = sqrt( xloc * xloc + yloc * yloc );
388  double theta = atan2(yloc,xloc);
389 
390  double dqdr;
391  if( R >= 0.25*a && R <= a)
392  {
393  std::cout << " Nonzeo value !!! " << endl ;
394  dqdr = -4/(3*a);
395  }
396  else
397  dqdr = 0.0 ;
398  dqdx1 = dqdr * cos(theta);
399  dqdx2 = dqdr * sin(theta);
400  }*/
401 
402  // COMPUTE THE STRESS OF ELEM AT GAUSS POINT GP
403  epsilon = B -> Times(U) ;
404  sigma = D->Times(epsilon);
405 
406  // TRANSFORM THESE TERMS OF STATE (1) TO LOCAL CRACK TIP COORDINATE SYSTEM
407  // using formula : (rot_matrix)*H*(rot_matrix)'
408 
409  rdu1dx = r -> Times(dU1dx) ;
410  LocalGradDisp = rdu1dx -> Times(rt) ;
411 
412  FloatArray *LocalSigma = new FloatArray(3);
413  double C2T = cos(2.*alpha) ;
414  double S2T = sin(2.*alpha) ;
415 
416  (*LocalSigma)[0] = 0.5*((*sigma)[0]+(*sigma)[1])+
417  0.5*((*sigma)[0]-(*sigma)[1])*C2T + (*sigma)[2]*S2T ;
418 
419  (*LocalSigma)[1] = 0.5*((*sigma)[0]+(*sigma)[1])-
420  0.5*((*sigma)[0]-(*sigma)[1])*C2T - (*sigma)[2]*S2T ;
421 
422  (*LocalSigma)[2] = (*sigma)[2]*C2T - 0.5*((*sigma)[0] - (*sigma)[1])*S2T ;
423 
424  // COMPUTATION OF COMPONENTS OF THE AUXILIARY FIELD
425 
426  FloatArray AuxEps(3);
427  FloatArray AuxStress(3);
428  FloatMatrix AuxGradDisp(2,2) ;
429 
430  globalCoordGP = (*i)->giveFEInterpolation()->local2Global(domain,nodeArr,coord);
431 
432  // loop on modes
433  for(size_t u = 0 ; u < 2 ; u++)
434  {
435  if( matArray->size() == 1 ) // homogeneous media
436  {
437  if(u == 0)
438  auxFieldHomo->ComputeComponentsOfAuxField(this,globalCoordGP,Mode_i,AuxGradDisp,AuxEps,AuxStress);
439  else
440  auxFieldHomo->ComputeComponentsOfAuxField(this,globalCoordGP,Mode_ii,AuxGradDisp,AuxEps,AuxStress);
441  }
442  else // bimaterial media
443  {
444  if(u == 0)
445  {
446  auxFieldBiMat->ComputeComponentsOfAuxField(this,globalCoordGP,Mode_i,AuxGradDisp,AuxEps,AuxStress);
447  }
448  else
449  auxFieldBiMat->ComputeComponentsOfAuxField(this,globalCoordGP,Mode_ii,AuxGradDisp,AuxEps,AuxStress);
450  }
451 
452  // compute the interaction strain energy W(1,2)= sigma_{ij}*epsilon_{ij}
453  double W = (*LocalSigma)[0]*AuxEps.at(1) + 2.0*(*LocalSigma)[2]*AuxEps.at(3) +
454  (*LocalSigma)[1]*AuxEps.at(2) ;
455 
456  // computing terms in the interaction integral ...
457 
458  double I1 =
459  ( (*LocalSigma)[0]*AuxGradDisp.at(1,1)+(*LocalSigma)[2]*AuxGradDisp.at(2,1) ) * dqdx1 +
460  ( (*LocalSigma)[2]*AuxGradDisp.at(1,1)+(*LocalSigma)[1]*AuxGradDisp.at(2,1) ) * dqdx2 ;
461 
462  double I2 =
463  ( AuxStress.at(1) * LocalGradDisp->at(1,1) + AuxStress.at(3) * LocalGradDisp->at(2,1) ) * dqdx1 +
464  ( AuxStress.at(3) * LocalGradDisp->at(1,1) + AuxStress.at(2) * LocalGradDisp->at(2,1) ) * dqdx2 ;
465 
466  I[u] += (I1 + I2 - W*dqdx1) * dV ;
467  }
468 
469  delete dNdx ; delete dU1dx ; delete epsilon; delete sigma; delete LocalGradDisp;
470  delete B ; delete LocalSigma ; delete rdu1dx ;
471 
472  // export Gauss points to Matlab for plot ...
473  //fprintf (mlbFile,"%5.6f %5.6f \n",globalCoordGP->x,globalCoordGP->y);
474 
475  } // end of loop on Gauss points
476 
477  // delete GPs used for J integral
478  for (size_t j = 0 ; j < (*i)->giveNumOfGPtsForJ_Integral() ; j++)
479  delete gaussPointArray[j] ;
480  delete gaussPointArray ;
481  delete D;
482  } // end of loop on elements
483  //fprintf (mlbFile," ];\n ");
484  // -----------------------------------------------------------------------------------------------
485  // EXPORT MATLAB PLOT COMMANDS
486  /*
487  fprintf (mlbFile,"figure;\n ");
488  fprintf (mlbFile,"hold on ;\n") ;

529 double CrackTip::computeSIFMultiplierForOneMat()
530 // *********************************************
531 // Ki = this->computeSIFMultiplierForOneMat() * Interaction integral
532 {
533  this->giveMatArray();
534 
535  double E = (*matArray)[0]->give('E');
536  double nu = (*matArray)[0]->give('n');
537 
538  if (field == PlaneStress)
539  return (E /2.0 );
540  else if (field == PlaneStrain)
541  return E/(2.0 - nu*nu - nu*nu);
542  else
543  assert(false);
544 }

546 double CrackTip::computeSIFMultiplierForBiMat()
547 // ********************************************
548 // The same as CrackTip::computeSIFMultiplierForOneMat() but for
549 // interfacial cracks
550 {
551  this->giveMatArray();
552 
553  double E1 = (*matArray)[0]->give('E');
554  double nu1 = (*matArray)[0]->give('n');
555 
556  double E2 = (*matArray)[1]->give('E');
557  double nu2 = (*matArray)[1]->give('n');
558 
559  double E1bar,E2bar;
560  if (field == PlaneStress)
561  {
562  E1bar = E1;
563  E2bar = E2;
564  }
565  else if (field == PlaneStrain)
566  {
567  E1bar = E1/(1. - nu1 * nu1);
568  E2bar = E2/(1. - nu2 * nu2);
569  }
570  else
571  assert(false);
572 
573  double Eetoile = (E1bar*E2bar)/(E1bar+E2bar);
574 
575  double shear1 = E1/(1. + nu1 + nu1);
576  double shear2 = E2/(1. + nu2 + nu2);
577 
578  double K1 = Field<PlaneStrain>::K(nu1); // the Kolosov coeficient
579  double K2 = Field<PlaneStrain>::K(nu2);
580 
581  double epsilon = 0.5/M_PI*log((shear1 + shear2*K1)/(shear1*K2 + shear2));
582 
583  return Eetoile * cosh(M_PI*epsilon) * cosh(M_PI*epsilon);
584 }
585 double CrackTip::computeSIFMultiplier()
586 // *************************************
587 // K_i = this->computeSIFMultiplier()*this->computeInteractionIntegral()
588 {
589  if (tipID == HomoElast)
590  return this->computeSIFMultiplierForOneMat();
591  else if (tipID == BiMatElast)
592  return this->computeSIFMultiplierForBiMat();
593  else
