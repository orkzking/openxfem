45 void Tetra4 :: computeGaussPoints ()
46 // *********************************
47 // Sets up the array containing the Gauss points of the receiver.
48 // Three Gauss quadrature rules are used
49 // 1. Classical elements : 1 gauss points are used
50 // 2. Split and tip elements : 13 Gp for each sub-triangles
51 // 3. Near tip enriched elements : 13 GP are used (due to asymptotic functions)
52 // REMARK: \TODO for 3D X-FEM implementation !!!
53 {
54  numberOfIntegrationRules = 1 ;
55  quadratureRuleArray = new IntegrationRule*;
56 
57  // choose the correct integration rule to be used ...
58 
59  if (this->isEnriched() == false ) // classical elements
60  {
61  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
62  quadratureRuleArray[0]->setUpIntegrationPoints(::TETRAHEDRA,1);
63  }
64  else if (enrichmentItemListOfElem != NULL) // split, tip elements
65  {
66  quadratureRuleArray[0] = new SplitGaussLegendreQuadrature;
67  quadratureRuleArray[0]->setUpIntegrationPoints(this,13);
68  }
69  else // partially enriched elements
70  {
71  size_t count = 0 ;
72  for(size_t i = 0 ; i < numberOfNodes ; i++)
73  {
74  if(this->giveNode(i+1)->isTipEnriched())
75  count += 1 ;
76  }
77  if(count != 0) // near tip enriched elements
78  {
79  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
80  quadratureRuleArray[0]->setUpIntegrationPoints(::TETRAHEDRA,13);
81  }
82  else
83  {
84  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
85  quadratureRuleArray[0]->setUpIntegrationPoints(::TETRAHEDRA,3);
86  }
87  }
88 
89  // tranform the obtained integration points into the GaussPoint array of element
90 
91  vector<double>* weight = quadratureRuleArray[0]->giveWeightArray();
92  vector<Mu::Point*>* coordArray = quadratureRuleArray[0]->giveIntegrationPointVector();
93 
94  if(gaussPointArray)
95  {
96  for (size_t i = 0 ; i < numberOfGaussPoints ; i++)
97  delete gaussPointArray[i] ;
98  delete gaussPointArray ;
99  }
100 
101  numberOfGaussPoints = coordArray->size();
102 
103  gaussPointArray = new GaussPoint* [numberOfGaussPoints] ;
104 
105  for(size_t i = 0 ; i < numberOfGaussPoints ; i++)
106  {
107  gaussPointArray[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
108  }
109  delete weight ; delete coordArray ; // Purify, 08-11-05
110 }

142 GaussPoint** Tetra4 :: setGaussQuadForJ_Integral()
143 // ***********************************************
144 // Two Gauss quadrature rules are used
145 // 1. Classical elements : 7 Gauss points are used
146 // 2. Split and tip elements : 13 Gps for each sub-triangles
147 
148 {
149  IntegrationRule *GaussQuad = new IntegrationRule ;
150 
151  if (!enrichmentItemListOfElem) // no need discontinuous Gauss quad
152  {
153  GaussQuad = new StandardGaussLegendreQuadrature;
154  GaussQuad->setUpIntegrationPoints(::TRIANGLE,7);
155  }
156  else
157  {
158  GaussQuad = new SplitGaussLegendreQuadrature;
159  GaussQuad->setUpIntegrationPoints(this,7);
160  }
161 
162  // tranform the obtained integration points into the GaussPoint array of element
163 
164  vector<double> *weight = GaussQuad->giveWeightArray();
165  vector<Mu::Point*> *coordArray = GaussQuad->giveIntegrationPointVector();
166 
167  numOfGPsForJ_Integral = coordArray->size() ;
168 
169  GaussPoint** ret = new GaussPoint* [numOfGPsForJ_Integral] ;
170 
171  for(size_t i = 0 ; i < numOfGPsForJ_Integral ; i++)
172  {
173  ret[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
174  }
175 
176  delete weight ; delete coordArray ; // Purify, 14-10-05
177 
178  return ret ;
179 }
