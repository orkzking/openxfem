61 FloatMatrix* PlateIso :: ComputeBBmatrixAt (GaussPoint *aGaussPoint)
62 //********************************************************************
63 // Returns the [3x8] bending strain-displacement BB matrix of the receiver,
64 // evaluated at aGaussPoint.
65 // Format of BB = [0 0 N1,x ...
66 // 0 -N1,y 0 ...
67 // 0 -N1,x N1,y ]
68 {
69  Mu::Point *Coord = aGaussPoint -> giveCoordinates() ;
70  FloatMatrix *dNdx = this->giveFEInterpolation()->evaldNdx(domain,this->giveNodeArray(),Coord);
71 
72  FloatMatrix *answer = new FloatMatrix(3,3*numberOfNodes);
73 
74  for (size_t i = 0 ; i < numberOfNodes ; i++)
75  {
76  answer->at(1,3*i+3) = dNdx->at(i+1,1);
77  answer->at(2,2*i+2) = -dNdx->at(i+1,2);
78  answer->at(3,2*i+2) = -dNdx->at(i+1,1);
79  answer->at(3,3*i+3) = dNdx->at(i+1,2);
80  }
81 
82  delete dNdx ;
83 
84  return answer ;

87 FloatMatrix* PlateIso :: ComputeBSmatrixAt (GaussPoint *aGaussPoint)
88 //********************************************************************
89 // Returns the [2x8] shearing strain-displacement BB matrix of the receiver,
90 // evaluated at aGaussPoint.
91 // Format of BS = [N1,x 0 N1 ...
92 // N1,y -N1 0 ...]
93 {
94  Mu::Point *Coord = aGaussPoint -> giveCoordinates() ;
95  FloatMatrix *dNdx = this->giveFEInterpolation()->evaldNdx(domain,this->giveNodeArray(),Coord);
96  FloatArray *n = this->giveFEInterpolation()->evalN(Coord);
97 
98  FloatMatrix *answer = new FloatMatrix(2,3*numberOfNodes);
99 
100  for (size_t i = 0 ; i < numberOfNodes ; i++)
101  {
102  answer->at(1,2*i+1) = dNdx->at(i+1,1);
103  answer->at(1,3*i+3) = n->at(i+1);
104  answer->at(2,2*i+1) = dNdx->at(i+1,2);
105  answer->at(2,2*i+2) = -n->at(i+1);
106  }
107 
108  delete dNdx ;
109 
110  return answer ;
111 }

125  constitutiveMatrix->at(2,i) = Eb->at(2,i);
126  constitutiveMatrix->at(3,i) = Eb->at(3,i);
127  }
128 
129  constitutiveMatrix->at(4,4) = Eb->at(1,1);
130  constitutiveMatrix->at(5,5) = Eb->at(2,2);
131 
132  delete Eb ; delete Es ;
133  return constitutiveMatrix ;

136 FloatMatrix* PlateIso ::computeConstitutiveMatrixB()
137 // ***************************************************
138 // computes the constituive matrix coresponding to bending part
139 {
140  Material *mat = this -> giveMaterial() ;
141 
142  double h = mat -> give('t') ;
143  double e = mat -> give('E') ;
144  double nu = mat -> give('n') ;
145 
146  double D = e*h*h*h/(12.*(1.-nu*nu)); // flexural rigidity
147 
148  FloatMatrix *ret = new FloatMatrix(3,3);
149  ret->at(1,1) = D ; ret->at(1,2) = D*nu ;
150  ret->at(2,1) = D*nu ; ret->at(2,2) = D ;
151  ret->at(3,3) = 0.5*D*(1. - nu) ;
152 
153  return ret;
154 }

190 FloatMatrix* PlateIso :: computeBendingStiffnessMatrix()
191 //*******************************************************
192 // Computes numerically the bending stiffness matrix of the receiver, [Kb]
193 {
194  FloatMatrix *b,*db,*d ;
195  GaussPoint *gp ;
196  double dV ;
197  stiffnessMatrix = new FloatMatrix() ;
198  for(int i = 0 ; i < numberOfGaussPoints ; i++)
199  {
200  gp = gaussPointArray[i] ;
201  b = this -> ComputeBBmatrixAt(gp) ;
202  d = this -> computeConstitutiveMatrixB() ;
203  dV = this -> computeVolumeAround(gp) ;
204  db = d -> Times(b) ;
205  stiffnessMatrix -> plusProduct(b,db,dV) ;
206 
207  delete b ;
208  delete db ;
209  }
210 
211  return stiffnessMatrix -> symmetrized() ;

262 double PlateIso :: computeVolumeAround (GaussPoint* aGaussPoint)
263 // ***************************************************************
264 // Returns the portion of the receiver which is attached to aGaussPoint.
265 {
266  FloatMatrix* jacob ;
267  double determinant,weight,thickness,volume ;
268 
269  Mu::Point *coord = aGaussPoint -> giveCoordinates() ;
270  jacob = this -> giveFEInterpolation()->giveJacobianMatrixAt(domain,this->giveNodeArray(),coord);
271  determinant = fabs (jacob->giveDeterminant()) ;
272  weight = aGaussPoint -> giveWeight() ;
273  thickness = this -> giveMaterial() -> give('t') ;
274 
275  volume = determinant * weight * thickness ;
276 
277  delete jacob ;
278  return volume ;
279 }

282 void PlateIso :: computeGaussPoints()
283 // ************************************
284 // Sets up the array containing the Gauss points of the receiver.
285 // Modified to use class Integration Rule and discontinuous Gauss quadrature.
286 {
287  numberOfIntegrationRules = 1 ;
288  quadratureRuleArray = new IntegrationRule*;
289 
290  int numOfGPs = 16 ;
291  if (this->isEnriched() == false ) // classical elements
292  {
293  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
294  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
295  }
296  else if (enrichmentItemListOfElem != NULL) // split, tip elements
297  {
298  quadratureRuleArray[0] = new SplitGaussLegendreQuadrature;
299  quadratureRuleArray[0]->setUpIntegrationPoints(this,13);
300  }
301  else // partially enriched elements
302  {
303  size_t count = 0 ;
304  for(size_t i = 0 ; i < numberOfNodes ; i++)
305  {
306  if(this->giveNode(i+1)->isTipEnriched())
307  count += 1 ;
308  }
309  if(count != 0) // near tip enriched elements
310  {
311  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
312  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,numOfGPs);
313  }
314  else // step enriched elements
315  {
316  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
317  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
318  }
319  }
320 
321  // tranform the obtained integration points into the GaussPoint array of element
322 
323  vector<double>* weight = quadratureRuleArray[0]->giveWeightArray();
324  vector<Mu::Point*>* coordArray = quadratureRuleArray[0]->giveIntegrationPointVector();
325 
326  numberOfGaussPoints = coordArray->size();
327 
328  gaussPointArray = new GaussPoint* [numberOfGaussPoints] ;
329 
330  for(size_t i = 0 ; i < numberOfGaussPoints ; i++)
331  {
332  gaussPointArray[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
333  }
334  delete weight ; delete coordArray ; // Purify, 08-11-05
335 }
