56 void MaterialInterface::treatMeshGeoInteraction(Element *e)
57 // ********************************************************
58 // Check if element interacts with enrichment item or not. If so, insert this element
59 // into the list of each enrichment item
60 // Modified at 2005-09-07 to make it more efficient than before.
61 // 28-12-2005: MATERIAL INTERFACE IMPLEMENTATION, ASSUMING 2 MATERIALS
62 {
63  if( this->interactsWith(e) )
64  {
65  e->isEnrichedWith(this);
66  this->setListOfInteractedElements(e);
67  e->computeNodalLevelSets(this);
68  e->setMultiMaterial();
69  e->setMaterialIDs(this->giveMaterialIDs());
70  }
71  else
72  {
73  Mu::Point *p = e->giveNode(1)->makePoint();
74  double ls = this->giveMyGeo()->computeSignedDistanceOfPoint(p);
75  if(ls > 0)
76  e->setMaterial(this->giveMaterialIDs()[0]);
77  else
78  e->setMaterial(this->giveMaterialIDs()[1]);
79  }
80 
81 }
