95  // strain increment corresponding to dxacc = delta x accumulated

97  deltaEpsilon = elem->computeStrainIncrement(gp,dElem);
98 
99  // delta sigma
100  sigmaTrial = Del->Times(deltaEpsilon); // SC-Purify-10.09.97
101  // sigma(prev)+delta sigma
102  sigmaTrial->add(gp->givePreviousStressVector());
103  //hardening terms
104  alphaPrev = gp->givePreviousBackStressVector()->GiveCopy();
105  if (elem->giveDomain()->giveTimeIntegrationScheme()->giveCurrentStep()->giveNumber() == 1) {
106  radius = (this->give('k'))*sqrt(2.);
107  } else {
108  radius = gp->givePreviousYieldRadius();
109  }
110  // computation of f(sigmaTrial)
111  double fSigTr = this->computeYieldFunctionFor(sigmaTrial, alphaPrev, radius);
112 
113  // check if the trial stress is outside the yield surface
114  if (fSigTr > 0) {
115  // if true: plastic correction
116  double deltaGamma;
117  FloatArray *dFDSigma = this->computeDFDSigma(sigmaTrial, alphaPrev);
118  //consistency...
119  deltaGamma = fSigTr / ((dFDSigma->transposedTimes(Del->Times(dFDSigma))) + (2./3.)*H);
120  //correction on the stress state
121  sigmaTrial->minus((Del->Times(dFDSigma))->times(deltaGamma));
122  //correction on the hardening parameters
123  FloatArray *ksi = this->computeKsi(sigmaTrial, alphaPrev);
124  ksi->times(1. / ksi->computeTensorialNorm());
125  alphaPrev->minus((ksi->Times(deltaGamma*(2./3.)*H*(1.-b)))->negated());
126  radius += (deltaGamma*(2./3.)*H*b);
127  //tell the GaussPoint that it has reached the yield surface
128  gp->isPlastic();
129  gp->setDeltaGamma(deltaGamma);
130  delete dFDSigma;
131  delete ksi;
132  }
133 
134  gp->letStressVectorBe(sigmaTrial); //gp stores sigmaTrial (corrected), not a copy!
135  gp->letBackStressVectorBe(alphaPrev); //gp stores alphaPrev (corrected), not a copy!
136  gp->setYieldRadius(radius);
137 
138  // deletion of temporary arrays
139  delete Del;
140  delete deltaEpsilon;
