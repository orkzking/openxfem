42 std::valarray<double> ModifiedHomoCrackAsymp::computeRampFunctionAt(Mu::Point* p)
43 {
44  CrackInterior* myCrack =
45  dynamic_cast<CrackTip*>(this->activeEnrItem)->giveMyCrackInterior();
46 
47  double a = myCrack->giveCrackLength() ; // half crack length
48 
49  FloatArray * localCoord =
50  dynamic_cast<CrackTip*>(this->activeEnrItem)->computeLocalCoordOf(p) ;
51 
52  double xloc = (*localCoord)[0];
53  double yloc = (*localCoord)[1];
54 
55  // compute ramp function and its local derivatives
56  double ramp,dRdx1 ;
57  if(xloc <= -2.0*a)
58  {
59  ramp = 0.0 ;
60  dRdx1 = 0.0 ;
61  }
62  else if(xloc <= 0.0 && xloc >= -2.0*a)
63  {
64  ramp = 0.25*(xloc+2.0*a)*(xloc+2.0*a)*(a-xloc)/(a*a*a) ;
65  dRdx1 = -0.75*(xloc+2.0*a)*xloc/(a*a*a) ;
66  }
67  else
68  {
69  ramp = 1.0 ;
70  dRdx1 = 0.0 ;
71  }
72 
73  std::valarray<double> result(3);
74  result[0] = ramp ;
75  result[1] = dRdx1 ;
76  result[2] = 0.0 ;
77 
78  return result ;
79 }

81 double ModifiedHomoCrackAsymp::EvaluateYourSelfAt(Mu::Point* coord)
82 // **************************************************************
83 // compute the modified first branch function:
84 // \sqrt{r}sin(theta/2)*ramp function
85 // used to model crack nucleation, see Dolbow and Bellec, 2003 for details.
86 {
87  std::valarray<double> *localCoord =
88  dynamic_cast<CrackTip*>(this->activeEnrItem)->computePolarCoordOf(coord) ;
89 
90  double r = (*localCoord)[0];
91  double theta = (*localCoord)[1];
92 
93  // for sure, -Pi <= theta <= Pi
94  assert(theta <= M_PI && theta >= -M_PI);
95 
96  std::valarray<double> ramp = this->computeRampFunctionAt(coord);
97 
98  double ret = sqrt(r) * sin(theta/2.) * ramp[0] ;
99  return ret ;
100 }

102 FloatArray* ModifiedHomoCrackAsymp::EvaluateYourGradAt(Mu::Point* p)
103 // **************************************************************
104 // compute the grad of the modified asymp. function: phi1(x)*r(x)
105 // First, derivatives w.r.t local crack tip are computed. Then, transformed
106 // to the global coordinate system.
107 // result = [dPhidx dPhidy]
108 {
109  // local crack tip coordinates of Point p
110  FloatArray *localCoord =
111  static_cast<CrackTip*>(activeEnrItem)->computeLocalCoordOf(p) ;
112 
113  // compute the inclination angle of the local crack tip coord. sys w.r.t the x axe
114  Mu::Segment *seg = static_cast<CrackTip*>(activeEnrItem)->giveTipSegment();
115  double x1 = seg->first()->x ;
116  double y1 = seg->first()->y ;
117  double x2 = seg->second()->x ;
118  double y2 = seg->second()->y ;
119 
120  double alpha = atan2(y2-y1,x2-x1) ; // inclination angle
121 
122  double xloc = (*localCoord)[0];
123  double yloc = (*localCoord)[1];
124 
125  double r = sqrt( xloc * xloc + yloc * yloc );
126  double theta = atan2(yloc,xloc) ;
127 
128  // for sure, -Pi <= theta <= Pi
129  assert(theta <= M_PI && theta >= -M_PI);
130 
131  double r2 = sqrt(r) ; // r = 0 ???
132  double fac = 0.5/r2 ;
133 
134  // local derivatives of first branch function
135  double dPhi1dx1 = -fac * sin(theta/2.);
136  double dPhi1dx2 = fac * cos(theta/2.);
137 
138  // local derivatives of ramp function
139 
140  double Phi1 = sqrt(r) * sin(theta/2.) ;
141  std::valarray<double> ramp = this->computeRampFunctionAt(p);
142 
143  double dPhidx1 = dPhi1dx1 * ramp[0] + Phi1 * ramp[1] ;
144  double dPhidx2 = dPhi1dx2 * ramp[0] + Phi1 * ramp[2] ;
145 
146  FloatArray* answer = new FloatArray(2);
147  (*answer)[0] = dPhidx1 * cos(alpha) - dPhidx2 * sin(alpha) ;
148  (*answer)[1] = dPhidx1 * sin(alpha) + dPhidx2 * cos(alpha) ;
149 
150  return answer;
151 }
