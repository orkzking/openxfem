77 bool Vertex :: interactsWith(Element* elem)
78 // ****************************************
79 // check if element elem contains the Vertex using the geometry
80 // predicates : orientation test
81 {
82  size_t count = 0;
83  double const EPSILON = 1e-8 ; // this tolerence has strong effect on the tip !!!
84 
85  double x0 = this->giveCoordinate(1); // coord. of this vertex
86  double y0 = this->giveCoordinate(2);
87 
88  double delta,x1,y1,x2,y2 ;
89 
90  size_t numNodes = elem->giveNumberOfNodes();
91 
92  for (size_t i = 1 ; i <= numNodes ; i++)
93  {
94  // coordinates of first node of edge
95  x1 = elem->giveNode(i)->giveCoordinate(1);
96  y1 = elem->giveNode(i)->giveCoordinate(2);
97 
98  // coordinates of second node of edge
99  if(i != numNodes)
100  {
101  x2 = elem->giveNode(i+1)->giveCoordinate(1);
102  y2 = elem->giveNode(i+1)->giveCoordinate(2);
103  }
104  else
105  {
106  x2 = elem->giveNode(1)->giveCoordinate(1);
107  y2 = elem->giveNode(1)->giveCoordinate(2);
108  }
109 
110  delta = (x1-x0)*(y2-y0) - (x2-x0)*(y1-y0) ;
111 
112  if (delta > EPSILON)
113  count += 1 ;
114  }
115 
116  return (count == numNodes);
117 }

152 std::vector<Mu::Point> Vertex :: intersection(const Mu::Triangle* t)
153 // *****************************************************************
154 // two cases may occur :
155 // 1. t cuts the crack segment(normal one),then return the intersection point
156 // and the Vertex itself.
157 // 2. crack segment is entirely inside the triangle => !!! Corrected : 2005-09-23
158 {
159  std::vector<Mu::Point> ret ;
160  std::vector<Mu::Point> temp ;
161 
162  Mu::Segment * seg = myParent->giveSegmentContain(this);
163 
164  if(this->intersects(t)) // normal case 1
165  {
166  temp = seg->intersection(t) ;
167  ret.insert(ret.end(), temp.begin(),temp.end());
168  // check for case crack goes throught node. Corrected 2005-08-09 when deal with
169  // structured triangle mesh .
170  if(ret.size() == 0)
171  {
172  for(size_t i = 0 ; i < 3 ; i++)
173  {
174  if(seg->on(t->getBoundingPoint(i)))
175  ret.push_back(*(t->getBoundingPoint(i)));
176  }
177  }
178  }
179  else // crack segment within element. Occurred when crack grows with small increment.
180  {
181  std::vector<Mu::Point> intersect = myParent->intersection(t);
182  ret.insert(ret.end(),intersect.begin(),intersect.end());
183  }
184 
185  // insert the CrackTip into the result
186  Mu::Point point((*coordinates)[0],(*coordinates)[1]);
187  ret.push_back(point);
188 
189  if(ret.size() != 2)
190  {
191  std::cout <<" Total number of intersection points :" << ret.size() << endl;
192  assert(false);
193  }
194 
195  return ret ;
196 }
