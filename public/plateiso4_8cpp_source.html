39 void PlateIsoQ4 :: computeGaussPoints()
40 // ************************************
41 // Sets up the array containing the Gauss points of the receiver.
42 // Modified to use class Integration Rule and discontinuous Gauss quadrature.
43 {
44  numberOfIntegrationRules = 1 ;
45  quadratureRuleArray = new IntegrationRule*;
46  int numOfGPs = 16 ;
47  if (this->isEnriched() == false ) // classical elements
48  {
49  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
50  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
51  }
52  else if (enrichmentItemListOfElem != NULL) // split, tip elements
53  {
54  quadratureRuleArray[0] = new SplitGaussLegendreQuadrature;
55  quadratureRuleArray[0]->setUpIntegrationPoints(this,13);
56  }
57  else // partially enriched elements
58  {
59  size_t count = 0 ;
60  for(size_t i = 0 ; i < numberOfNodes ; i++)
61  {
62  if(this->giveNode(i+1)->isTipEnriched())
63  count += 1 ;
64  }
65  if(count != 0) // near tip enriched elements
66  {
67  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
68  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,numOfGPs);
69  }
70  else // step enriched elements
71  {
72  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
73  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
74  }
75  }
76 
77  // tranform the obtained integration points into the GaussPoint array of element
78 
79  vector<double>* weight = quadratureRuleArray[0]->giveWeightArray();
80  vector<Mu::Point*>* coordArray = quadratureRuleArray[0]->giveIntegrationPointVector();
81 
82  numberOfGaussPoints = coordArray->size();
83 
84  gaussPointArray = new GaussPoint* [numberOfGaussPoints] ;
85 
86  for(size_t i = 0 ; i < numberOfGaussPoints ; i++)
87  {
88  gaussPointArray[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
89  }
90  delete weight ; delete coordArray ; // Purify, 08-11-05
91 }

101 GaussPoint** PlateIsoQ4 :: setGaussQuadForJ_Integral()
102 // ************************************************
103 // Sets up the array containing the Gauss points used for the J integral
104 // computation.
105 {
106 
107  IntegrationRule *GaussQuad = new IntegrationRule ;
108  size_t numOfGPs1 = 9 ;
109  size_t numOfGPs2 = 7 ; // number of GPs for each sub-triangle
110 
111  // choose the correct integration rule to be used ...
112 
113  if (!enrichmentItemListOfElem) // no enrichment
114  {
115  GaussQuad = new StandardGaussLegendreQuadrature;
116  GaussQuad->setUpIntegrationPoints(SQUARE,numOfGPs1); // 9,25,36,49,56,
117  }
118  else
119  {
120  GaussQuad = new SplitGaussLegendreQuadrature;
121  GaussQuad->setUpIntegrationPoints(this,numOfGPs2); // 7 or 13
122  }
123 
124  vector<double>* weight;
125  vector<Mu::Point*>* coordArray;
126 
127  weight = GaussQuad->giveWeightArray();
128  coordArray = GaussQuad->giveIntegrationPointVector();
129 
130  numOfGPsForJ_Integral = coordArray->size() ;
131  GaussPoint** ret = new GaussPoint* [numOfGPsForJ_Integral] ;
132 
133  for(size_t i = 0 ; i < weight->size() ; i++)
134  ret[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
135 
136  delete weight ; delete coordArray ; // Purify, 14-10-05
137 
138  return ret ;
139 }

160 double PlateIsoQ4 :: area()
161 {
162  double x0 = this->giveNode(1)->giveCoordinate(1);
163  double y0 = this->giveNode(1)->giveCoordinate(2);
164 
165  double x1 = this->giveNode(2)->giveCoordinate(1);
166  double y1 = this->giveNode(2)->giveCoordinate(2);
167 
168  double x2 = this->giveNode(3)->giveCoordinate(1);
169  double y2 = this->giveNode(3)->giveCoordinate(2);
170 
171  double x3 = this->giveNode(4)->giveCoordinate(1);
172  double y3 = this->giveNode(4)->giveCoordinate(2);
173 
174  double A1 = 0.5 * ((x0-x2)*(y1-y2) - (x1-x2)*(y0-y2)) ; // signed area
175  double A2 = 0.5 * ((x0-x3)*(y2-y3) - (x2-x3)*(y0-y3)) ; // signed area
176 
177  return A1 + A2 ;
178 }
