37  double ls; double ret = 0.0 ;

41  if(ls == 0) // signed distance not yet computed at this node

43  ls = activeEnrItem->giveMyGeo()->computeSignedDistanceOfPoint(e->giveNode(i+1)->makePoint());

45  }
