32 bool LevelSetDescription::interactsWith(Element *elem, GeometryEntity* geo)
33 // ************************************************************************
34 // computes the LS(signed distance) at nodes of elem, if at least there are
35 // two LS with different signs => elem cut by geo, see Stolarska 2001.
36 {
37  double dMax,dMin;
38  Mu::Point* coord = new Mu::Point();
39  std::vector<double> distVector;
40 
41 
42  for (size_t i=0 ; i< elem->giveNumberOfNodes() ; i++)
43  {
44  coord->x = elem->giveNode(i+1)->giveCoordinate(1);
45  coord->x = elem->giveNode(i+1)->giveCoordinate(2);
46  distVector.push_back(geo->computeSignedDistanceOfPoint(coord)) ;
47  }
48 
49  typedef std::vector<double>::iterator vec_iter ;
50 
51  vec_iter start,end;
52 
53  start = distVector.begin();
54  end = distVector.end();
55 
56  dMax = *(max_element(start,end));
57  dMin = *(min_element(start,end));
58 
59  return (dMax*dMin<0)? true : false ;
60 }
