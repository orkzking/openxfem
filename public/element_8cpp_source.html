397 FloatMatrix* Element :: ComputeNewmarkLhsAt (TimeStep* stepN)
398 // Computes the contribution of the receiver to the left-hand side of the
399 // linear system, using Newmark's formula.
400 {
401  FloatMatrix *m,*lhs ;
402  double beta,dt ;
403 
404  if (stepN->giveNumber() == 0) {
405  lhs = this -> GiveStiffnessMatrix() ;
406  } else {
407  beta = domain -> giveTimeIntegrationScheme() -> giveBeta() ;
408  if (beta == 0.0)
409  {
410  printf ("Error: beta = 0.0 in Newmark \n") ;
411  exit(0) ;
412  }
413  else
414  {
415  dt = stepN -> giveTimeIncrement() ;
416  m = this -> giveMassMatrix() -> Times(1.0 / (beta*dt*dt));
417  lhs = this -> GiveStiffnessMatrix();
418  lhs->plus(m);
419  delete m;
420  }
421  }
422 
423  return lhs ;
424 }

426 FloatArray* Element :: ComputeNewmarkRhsAt (TimeStep* stepN, FloatArray* dxacc)
427 // Computes the contribution of the receiver to the right-hand side of the
428 // linear system, using Newmark's formula.
429 {
430  FloatMatrix *K;
431  DiagonalMatrix *M;
432  FloatArray *fExt,*dPred,*rhs,*a,*d,*dElem,*dPrev,*temp;
433  double beta,dt ;
434 
435  fExt = this -> ComputeLoadVectorAt(stepN) ;
436 
437  if (stepN->giveNumber() == 0)
438  {
439  // computes also the true stress state at the intial step, through the internal
440  // forces computation.
441  K = this -> GiveStiffnessMatrix () ;
442  d = this -> ComputeVectorOf ('d',stepN) ;
443  dElem = dxacc->Extract(this->giveLocationArray());
444  rhs = K -> Times(d) -> add(fExt) -> add (this->ComputeInternalForces(dElem)->negated());
445  delete d;
446  delete dElem;
447  delete K;
448  }
449  else
450  {
451  dPred = this -> ComputeVectorOf ('D',stepN) ;
452  dPrev = this -> ComputeVectorOf ('d',domain -> giveTimeIntegrationScheme() -> givePreviousStep()) ;
453  double aNorm = dxacc->giveNorm();
454  if (aNorm == 0.)
455  { //means iteration zero (dxacc = 0)
456  dElem = dPred->Minus(dPrev);
457  rhs = (this->ComputeInternalForces(dElem)->negated()) -> add(fExt);
458  delete dElem;
459  } else
460  {
461  M = (DiagonalMatrix*) this -> giveMassMatrix () ;
462  dt = stepN -> giveTimeIncrement() ;
463  beta = domain -> giveTimeIntegrationScheme() -> giveBeta() ;
464  dElem = dxacc->Extract(this->giveLocationArray());
465  a = dElem->Times(1.0 / (beta*dt*dt));
466  temp = dPred->Minus(dPrev);
467  temp->add(dElem);
468  rhs = (M->Times(a->negated())) -> add(this->ComputeInternalForces(temp)->negated()) -> add(fExt);
469  delete a ;
470  delete dElem;
471  delete temp;
472  }
473  delete dPred;
474  delete dPrev;
475  }
476  delete fExt ;
477  return rhs ;
478 }

1065 FloatMatrix* Element ::ComputeBmatrixAt(GaussPoint *aGausspoint)
1066 // **************************************************************
1067 // Computes the general B matrix of the receiver, B = [Bu Ba]
1068 // Including non enriched part and enriched parts if element is enriched
1069 // B = [Bu Ba] with
1070 // Bu = [N1,x 0 N2,x 0 N3,x 0
1071 // 0 N1,y 0 N2,y 0 N3,y
1072 // N1,y N1,x N2,y N2,x N3,y N3,x]
1073 //
1074 // Ba = [(Nbar1(phi-phiAtNode1)),x 0 (Nbar2(phi-phiAtNode2)),x 0 (Nbar3(phi-phiAtNode3)),x 0
1075 // 0(Nbar1(phi-phiAtNode1)),y 0 (Nbar2(phi-phiAtNode2)),y 0 (Nbar3(phi-phiAtNode3)),y
1076 // (Nbar1(phi-phiAtNode1)),y (Nbar1(phi-phiAtNode1)),x ...]
1077 {
1078  // computes the standard part of B matrix : Bu
1079  FloatMatrix *Bu = this->ComputeBuMatrixAt(aGausspoint);
1080 
1081  // non enriched elements
1082  if (this->isEnriched() == false)
1083  return Bu;
1084 
1085  // enriched elements,then computes the enriched part Ba
1086  FloatMatrix *Ba = new FloatMatrix();
1087  vector<EnrichmentFunction*> *enrFnVector; // vector of enrichment functions
1088  FloatArray *gradPhiGP ; // grad of enr. func. at Gauss points
1089  FloatMatrix *temp ;
1090  double N,dNdx,dNdy,phiGP,phiNode,dPhidXGP,dPhidYGP ;
1091 
1092  Mu::Point *Coord = aGausspoint -> giveCoordinates() ; // local coord. of Gauss point
1093 
1094  // Get the shape functions multiplied with the enr. functions...
1095  FloatArray *Nbar = this->giveXFEInterpolation()->evalN(Coord);
1096  FloatMatrix *gradNbar = this->giveXFEInterpolation()->evaldNdx(domain,this->giveNodeArray(),Coord);
1097 
1098  for(size_t i = 0 ; i < numberOfNodes ; i++)
1099  {
1100  Node* nodeI = this->giveNode(i+1) ;
1101  if (nodeI->getIsEnriched()) // this node is enriched, then continue ...
1102  {
1103  N = (*Nbar)[i]; // shape function Ni
1104  dNdx = gradNbar->at(i+1,1) ; // derivative of Ni w.r.t x coord.
1105  dNdy = gradNbar->at(i+1,2) ; // derivative of Ni w.r.t y coord.
1106 
1107  // loop on enrichment items of nodeI
1108  list<EnrichmentItem*> *enrItemList = nodeI->giveEnrItemListOfNode();
1109  for (list<EnrichmentItem*>::iterator iter = enrItemList->begin(); iter != enrItemList->end(); ++iter)
1110  {
1111  // get the enrichment funcs of current enr. item
1112  enrFnVector = (*iter)->giveEnrFuncVector();
1113 
1114  //loop on vector of enrichment functions ...
1115  for(size_t k = 0 ; k < enrFnVector->size() ; k++ )
1116  {
1117  EnrichmentFunction* enrFn = (*enrFnVector)[k];
1118 
1119  // let Enr. function,enrFn, know for which enr. item it is modeling
1120  enrFn->findActiveEnrichmentItem(*iter);
1121 
1122  // value of enrichment function at gauss point
1123  phiGP = enrFn->EvaluateYourSelfAt(aGausspoint);
1124 
1125  // value of enrichment function at node
1126  phiNode = enrFn->EvaluateYourSelfAt(this,nodeI);
1127 
1128  // grad of enrichment function at Gauss point
1129  gradPhiGP = enrFn->EvaluateYourGradAt(aGausspoint);
1130  dPhidXGP = (*gradPhiGP)[0] ; // derivative of Phi w.r.t x coord.
1131  dPhidYGP = (*gradPhiGP)[1] ; // derivative of Phi w.r.t y coord.
1132 
1133  double a = dNdx * (phiGP - phiNode) + N * dPhidXGP ;
1134  double b = dNdy * (phiGP - phiNode) + N * dPhidYGP ;
1135 
1136  temp = new FloatMatrix(4,2);
1137 
1138  temp->at(1,1) = a ; temp->at(1,2) = 0.0 ;
1139  temp->at(2,1) = 0.0 ; temp->at(2,2) = b ;
1140  temp->at(3,1) = b ; temp->at(3,2) = a ;
1141 
1142  Ba = Ba->FollowedBy(temp);
1143  delete temp ; // Purify, 11-10-05
1144  delete gradPhiGP ; // Purify, 11-10-05
1145 
1146  } // end of loop on enr. functions
1147  } // end of loop on enr. items
1148  }
1149  } // end of loop on element nodes
1150 
1151  FloatMatrix *answer = Bu->FollowedBy(Ba);
1152 
1153  delete Ba ; // Purify, 11-10-05
1154  delete Nbar ; delete gradNbar ; // Purify, 11-10-05
1155 
1156  return answer;
1157 }
