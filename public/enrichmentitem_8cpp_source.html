81 EnrichmentItem* EnrichmentItem :: ofType (char* aClass)
82 // *****************************************************
83 // Returns a new enrichment item, which has the same number than the receiver,
84 // but belongs to aClass (Crack, or Hole).
85 {
86  EnrichmentItem* newEnrichItem ;
87 
88  if (! strcmp(aClass,"CrackInterior"))
89  newEnrichItem = new CrackInterior(number,domain) ;
90  else if (! strcmp(aClass,"CrackTip"))
91  newEnrichItem = new CrackTip(number,domain) ;
92  else if (! strcmp(aClass,"MaterialInterface"))
93  newEnrichItem = new MaterialInterface(number,domain) ;
94  else if (! strcmp(aClass,"Hole"))
95  newEnrichItem = new Hole(number,domain) ;
96  else {
97  printf ("%s : unknown enrichment item \n",aClass) ;
98  exit(0) ;}
99 
100  return newEnrichItem ;
101 }

117 vector<EnrichmentFunction*>* EnrichmentItem::giveEnrFuncVector()
118 // *************************************************************
119 // Reads from the input file the enrichment functions used to

121 {

123 
124  if (!myEnrichFns)
125  {
126  myEnrichFns = new vector<EnrichmentFunction*> ;
127  size_t size = this -> readInteger("EnrichmentFuncs") ;
128  for (size_t i = 0 ; i < size ; i++)
129  {
130  enrFn = domain->giveEnrichmentFunction(this->readInteger("EnrichmentFuncs",i+2));
131  enrFn->setMyEnrichmentItem(this) ; // enrFn used to model this enr. item
132  myEnrichFns->push_back(enrFn) ;
133  }
134  }
135 
136  return myEnrichFns ;

139 EnrichmentDetector* EnrichmentItem :: defineMyEnrDetector()
140 // *********************************************************
141 // Returns the EnrichmentDetector of the receiver. Creates it it does not exist yet
142 // Convention :
143 // enrichScheme = 1 : for CrackTip, enriche all nodes of tip-element
144 // enrichScheme = 2 : for CrackTip, enriche all nodes within a ball of radius r
145 // enrichScheme = 3 : for CrackInterior, enriche all nodes of split-element
146 // enrichScheme = 4 : for biofilms applications. For you, Ravindra :)
147 {
148  size_t enrDetectorID = this -> readInteger("enrichScheme");
149 
150  EnrichmentDetector* myEnrDetector;
151 
152  switch (enrDetectorID)
153  {
154  case 1:
155  myEnrDetector = new ElementAroundPointDetect ;
156  break;
157  case 2:
158  myEnrDetector = new BallAroundPointDetect ;
159  break;
160  case 3:
161  myEnrDetector = new SplitElementDetect ;
162  break;
163  case 4:
164  myEnrDetector = new BandOfWidthDetect ;
165  break;
166  }
167 
168  return myEnrDetector ;
169 }
