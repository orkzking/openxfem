9 FloatArray* FEI2dTriQuad :: evalN(Mu::Point* p)
10 // *********************************************
11 {
12  FloatArray* answer = new FloatArray(6) ;
13 
14  double ksi = p->x ;
15  double eta = p->y ;
16 
17  (*answer)[0] = 1 - 3*(ksi + eta) + 4*ksi*eta + 2*(ksi*ksi + eta*eta);
18  (*answer)[1] = ksi * (2*ksi-1);
19  (*answer)[2] = eta * (2*eta-1);
20  (*answer)[3] = 4* ksi * (1-ksi-eta);
21  (*answer)[4] = 4* ksi * eta;
22  (*answer)[5] = 4* eta * (1-ksi-eta);
23
