41 void SplitGaussLegendreQuadrature::SetUpPointsOn2dDomain(Element* elem,size_t nPoints)
42 // ************************************************************************************
43 // Computes the Gauss points for partitioned element
44 // using the sub-triangles got from the method Element::PartitionMySelf()
45 {
46  // get the subtriangles which already defined in local coordinate system
47 
48  std::vector<DelaunayTriangle *> * tri = elem->PartitionMySelf();
49 
50  /*// for debugging ...
51  std::cout << " This element is partitioned into " << tri->size() << " sub-triangles " << std::endl;
52  for(size_t i = 0 ; i < tri->size() ; i++)
53  {
54  (*tri)[i]->print() ;
55  std::cout << (*tri)[i]->area() << endl ;
56  } */
57 
58  // get Gauss points for each sub-triangles
59  StandardGaussLegendreQuadrature* standardQuad = new StandardGaussLegendreQuadrature;
60  standardQuad->setUpIntegrationPoints(::TRIANGLE,nPoints);
61 
62  std::vector<double>* weight;
63  std::vector<Mu::Point*>* coordArray;
64 
65  weight = standardQuad->giveWeightArray();
66  coordArray = standardQuad->giveIntegrationPointVector();
67 
68  // initialization ...
69  weightArray = new std::vector<double> ;
70  integrationPointVector = new std::vector<Mu::Point*> ;
71  FEI2dTriLin* FEInterpol = new FEI2dTriLin(1,2);
72 
73  FloatMatrix* coord ;
74  // ********************************************************************************
75  // ****** LOOP ON SUB-TRIANGLES ******
76  // ********************************************************************************
77  for(size_t j = 0 ; j < tri->size() ; j++ )
78  {
79  if( ((*tri)[j]->isTriangle) && ((*tri)[j]->area() > 1e-4 ) ) // new !!! 2005-09-13
80  {
81  coord = new FloatMatrix(3,2);
82  coord->at(1,1) = (*tri)[j]->first->x; // coord. of first vertex
83  coord->at(1,2) = (*tri)[j]->first->y;
84 
85  coord->at(2,1) = (*tri)[j]->second->x; // coord. of second vertex
86  coord->at(2,2) = (*tri)[j]->second->y;
87 
88  coord->at(3,1) = (*tri)[j]->third->x; // coord. of third vertex
89  coord->at(3,2) = (*tri)[j]->third->y;
90 
91  double area = (*tri)[j]->area(); // area of subtriangle
92 
93  // ********************************************************************************
94  // ****** LOOP ON GAUSS POINTS OF EACH SUB-TRIANGLE ******
95  // ********************************************************************************
96  for (size_t i = 0 ; i < coordArray->size() ; i++)
97  {
98 
99 #ifdef NVP5JULY05
100  std::cout << " " << " - the " << i+1 << " Gauss point : " << std::endl;
101  std::cout << " " ;
102  (*coordArray)[i]->print();
103  std::cout << std::endl;
104 #endif
105 
106  // transform this Gauss point to the global coord : \sum{N_i*x_i}
107  Mu::Point* newIntPoint = new Mu::Point();
108  FloatArray *temp ;
109  temp = FEInterpol->evalN((*coordArray)[i]);
110 
111  for (size_t j = 0 ; j < 3 ; j++)
112  {
113  double x = coord->at(j+1,1) ;
114  double y = coord->at(j+1,2) ;
115  newIntPoint->x += (*temp)[j]*x;
116  newIntPoint->y += (*temp)[j]*y;
117  }
118  delete temp; // Purify, 11-10-05
119 
120  double newWeight = 2.0 * weight->at(i) * area; // !!! 2.0 *
121 
122  integrationPointVector->push_back(newIntPoint);
123  weightArray->push_back(newWeight);
124 
125 #ifdef NVP5JULY05
126  for (size_t i = 0 ; i < integrationPointVector->size() ; i++)
127  {
128  std::cout << " " << " - the new " << i+1 << " Gauss point : " << std::endl;
129  std::cout << " " ;
130  (*integrationPointVector)[i]->print();
131  std::cout << std::endl;
132  }
133 #endif
134  } // end of loop on GPs
135  delete coord ; // Purify, 11-10-05
136  }
137  } // end of loop on sub-triangles
138 
139  delete weight ; delete coordArray ; // Purify, 11-10-05
140  delete FEInterpol ; // Purify, 11-10-05
141  delete tri ;
142 }
