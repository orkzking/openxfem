60 void BallAroundPointDetect :: setEnrichedNodes(EnrichmentItem *enrItem)
61 // ********************************************************************
62 // enrItem must be CrackTip !!!
63 // nodes belong to the circle are enriched (Fixed area enrichment).
64 // NVP 2005-06-03
65 {
66  // checking for compatibility ...
67  if (!(typeid(static_cast<CrackTip*>(enrItem)) == typeid(CrackTip*)))
68  {
69  std :: cout << " Enrichment item and Enrichment detector is not compatible together "<< std :: endl ;
70  assert(false);
71  }
72 
73  double enrichRadius = static_cast<CrackTip*>(enrItem)->giveEnrichRadius();
74  double xTip = static_cast<CrackTip*>(enrItem)->giveTip()->giveCoordinate(1);
75  double yTip = static_cast<CrackTip*>(enrItem)->giveTip()->giveCoordinate(2);
76  Mu::Point *center = new Mu::Point(xTip,yTip);
77  Mu::Circle *ball = new Mu::Circle(enrichRadius,center);
78 
79  // find nodes belong to the ball ...
80  vector<Element*> *interactedElements = enrItem->giveElementsInteractWithMe();
81  Element *tipElem = (*interactedElements)[0] ;
82 
83  std::list<Element*> ret = tipElem->conflicts(ball);
84 
85  Node *node;
86  Mu::Point *poi ;
87  for(std::list<Element*> :: iterator i = ret.begin() ; i != ret.end() ; i++)
88  {
89  for(size_t j = 0 ; j < (*i)->giveNumberOfNodes() ; j++)
90  {
91  node = (*i)->giveNode(j+1);
92  poi = node->makePoint();
93  if(ball->in(poi)) // node belongs to the ball, then enriched...
94  node->isEnrichedWith(enrItem);
95  delete poi ; //Purify,14-10-2005
96  }
97  }
98  delete center ; delete ball ; //Purify,14-10-2005
99 }
