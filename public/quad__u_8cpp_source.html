47 void Quad4_U :: computeGaussPoints ()
48 // **********************************
49 // Sets up the array containing the Gauss points of the receiver.
50 // Modified to use class Integration Rule and discontinuous Gauss quadrature.
51 {
52  numberOfIntegrationRules = 1 ;
53  quadratureRuleArray = new IntegrationRule*;
54 
55  int numOfGPs = 16 ;
56 
57  if (this->isEnriched() == false ) // classical elements
58  {
59  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
60  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
61  }
62  else if (enrichmentItemListOfElem != NULL) // split, tip elements
63  {
64  quadratureRuleArray[0] = new SplitGaussLegendreQuadrature;
65  quadratureRuleArray[0]->setUpIntegrationPoints(this,13);
66  }
67  else // partially enriched elements
68  {
69  size_t count = 0 ;
70  for(size_t i = 0 ; i < numberOfNodes ; i++)
71  {
72  if(this->giveNode(i+1)->isTipEnriched())
73  count += 1 ;
74  }
75  if(count != 0) // near tip enriched elements
76  {
77  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
78  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,numOfGPs);
79  }
80  else // step enriched elements
81  {
82  quadratureRuleArray[0] = new StandardGaussLegendreQuadrature;
83  quadratureRuleArray[0]->setUpIntegrationPoints(SQUARE,4);
84  }
85  }
86 
87  // tranform the obtained integration points into the GaussPoint array of element
88 
89  vector<double>* weight = quadratureRuleArray[0]->giveWeightArray();
90  vector<Mu::Point*>* coordArray = quadratureRuleArray[0]->giveIntegrationPointVector();
91 
92  numberOfGaussPoints = coordArray->size();
93 
94  gaussPointArray = new GaussPoint* [numberOfGaussPoints] ;
95 
96  for(size_t i = 0 ; i < numberOfGaussPoints ; i++)
97  {
98  gaussPointArray[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
99  }
100  delete weight ; delete coordArray ; // Purify, 08-11-05
101 }

103 GaussPoint** Quad4_U :: setGaussQuadForJ_Integral()
104 // ************************************************
105 // Sets up the array containing the Gauss points used for the J integral
106 // computation.
107 {
108  IntegrationRule *GaussQuad = new IntegrationRule ;
109 
110  size_t numOfGPs1 = 9 ;
111  size_t numOfGPs2 = 7 ; // number of GPs for each sub-triangle
112 
113  // choose the correct integration rule to be used ...
114 
115  if (!enrichmentItemListOfElem) // no enrichment
116  {
117  GaussQuad = new StandardGaussLegendreQuadrature;
118  GaussQuad->setUpIntegrationPoints(SQUARE,numOfGPs1); // 9,25,36,49,56,
119  }
120  else
121  {
122  GaussQuad = new SplitGaussLegendreQuadrature;
123  GaussQuad->setUpIntegrationPoints(this,numOfGPs2); // 7 or 13
124  }
125 
126  vector<double>* weight;
127  vector<Mu::Point*>* coordArray;
128 
129  weight = GaussQuad->giveWeightArray();
130  coordArray = GaussQuad->giveIntegrationPointVector();
131 
132  numOfGPsForJ_Integral = coordArray->size() ;
133  GaussPoint** ret = new GaussPoint* [numOfGPsForJ_Integral] ;
134 
135  for(size_t i = 0 ; i < weight->size() ; i++)
136  ret[i] = new GaussPoint(this,i+1,coordArray->at(i),weight->at(i),4) ;
137 
138  delete weight ; delete coordArray ; // Purify, 14-10-05
139 
140  return ret ;
141 }

167 std::vector<DelaunayTriangle *>* Quad4_U:: PartitionMySelf()
168 // ********************************************************
169 // 2005-08-23.
170 {
171  std::vector<Mu::Point> temp,pts,intersectPoints ;
172  GeometryEntity* myGeo;
173  // TIP-ELEMENTS INTERACTED WITH BOTH THE TIP AND THE CRACK INTERIOR,
174  // THEREFORE, NEED TO REMOVE CRACK INTERIOR FROM THE LIST OF ENRICHMENT ITEMS.
175 
176  list<EnrichmentItem*> ::iterator iter1,iter2;
177  iter1=find_if(enrichmentItemListOfElem->begin(),enrichmentItemListOfElem->end(),IsType<CrackTip,EnrichmentItem>());
178  iter2=find_if(enrichmentItemListOfElem->begin(),enrichmentItemListOfElem->end(),IsType<CrackInterior,EnrichmentItem>());
179 
180  if (iter1 != enrichmentItemListOfElem->end() && iter2 != enrichmentItemListOfElem->end())
181  enrichmentItemListOfElem->remove(*iter2);
182 
183  // LOOP ON ENRICHMENT ITEMS AND GET THE INTERSECTION POINTS
184  for(std::list<EnrichmentItem*>::iterator i = enrichmentItemListOfElem->begin() ; i != enrichmentItemListOfElem->end() ; i++)
185  {
186  if( (typeid(**i) == typeid(CrackTip)) && (dynamic_cast<CrackTip*>(*i)->amIOnElementEdge()) )
187  {
188  intersectPoints.push_back(*this->giveMyCenter());
189  }
190  else
191  {
192  myGeo = (*i)->giveMyGeo();
193  temp = myGeo->intersection(this);
194  if(temp.size() == 1) // for the case crack touched the element edge
195  {
196  Mu::Point *p1 = this->giveNode(1)->makePoint();
197  Mu::Point *p2 = this->giveNode(2)->makePoint();
198  Mu::Point *p3 = this->giveNode(3)->makePoint();
199  Mu::Point *p4 = this->giveNode(4)->makePoint();
200  Mu::Segment *s1 = new Mu::Segment(*p1,*p2) ;
201  Mu::Segment *s2 = new Mu::Segment(*p2,*p3) ;
202  Mu::Segment *s3 = new Mu::Segment(*p3,*p4) ;
203  Mu::Segment *s4 = new Mu::Segment(*p4,*p1) ;
204 
205  PiecewiseLinear *geo = dynamic_cast<PiecewiseLinear*>(myGeo);
206  std::list<Mu::Point*> *vertices = geo->giveMyListOfVertices();
207 
208  for (std::list<Mu::Point*> ::iterator k = vertices->begin() ; k != vertices->end(); k++)
209  {
210  if( s1->on(*k) || s2->on(*k) || s3->on(*k)|| s4->on(*k) )
211  {
212  (*k)->print();
213  temp.push_back(**k) ;
214  }
215  }
216  delete s1 ; delete s2 ; delete s3 ; delete s4 ;
217  delete p1 ; delete p2 ; delete p3 ; delete p4 ; // Purify, 15-10-2005
218  }
219 
220  intersectPoints.insert(intersectPoints.end(),temp.begin(),temp.end());
221 
222  // ************************************************************************************
223  // ******** LOOK FOR KINK POINT ******
224  // ************************************************************************************
225  // just segments have kink points, i.e., geometry of CrackInterior only
226  // So, use typeid() to test ...
227  // Attention: just greater than 2 vertices => kink point :) 2005-09-13
228 
229  if( typeid(**i) == typeid(CrackInterior) )
230  {
231  std::list<Mu::Point*> *vertices = dynamic_cast<PiecewiseLinear*>(myGeo)->giveMyListOfVertices();
232  if(vertices->size() != 2)
233  {
234  for(std::list<Mu::Point*> ::iterator k = vertices->begin() ; k != vertices->end(); k++)
235  {
236  if( this->isWithinMe(*k) )
237  intersectPoints.push_back(**k) ;
238  }
239  }
240  } // end of looking for kink points */
241  } // end of check CrackTip
242  } // end of loop on enrichment items
243 
244  //for(size_t i = 0 ; i < intersectPoints.size() ; i++)
245  //{
246  // intersectPoints[i].print();
247  // std::cout<<std::endl;
248  //}
249 
250  // make the four nodes Mu::Point
251  Mu::Point *p0 = new Mu::Point(this->giveNode(1)->giveCoordinate(1),this->giveNode(1)->giveCoordinate(2)) ;
252  Mu::Point *p1 = new Mu::Point(this->giveNode(2)->giveCoordinate(1),this->giveNode(2)->giveCoordinate(2)) ;
253  Mu::Point *p2 = new Mu::Point(this->giveNode(3)->giveCoordinate(1),this->giveNode(3)->giveCoordinate(2)) ;
254  Mu::Point *p3 = new Mu::Point(this->giveNode(4)->giveCoordinate(1),this->giveNode(4)->giveCoordinate(2)) ;
255 
256  // transform to local coordinate of the parent element
257  Mu::Point *localP0 = this->giveFEInterpolation()->global2Local(domain,this->giveNodeArray(),p0);

261 
262  DelaunayTree *dt = new DelaunayTree(localP0,localP1,localP2) ;
263  dt->insert(localP3);
264  // insert the intersection points into the DT
265  for(size_t j = 0 ; j < intersectPoints.size() ; j++)
266  {
267  dt->insert(standardFEInterpolation->global2Local(domain,this->giveNodeArray(),&intersectPoints[j]));
268  }
269 
270  // do the Delaunay triangulation ...
271  vector<DelaunayTriangle *> *tri = dt->getTriangles();
272 
273  delete p0 ; delete p1 ; delete p2 ; delete p3 ;
274 
275  return tri;

279 double Quad4_U :: area()
280 {
281  double x0 = this->giveNode(1)->giveCoordinate(1);
282  double y0 = this->giveNode(1)->giveCoordinate(2);
283 
284  double x1 = this->giveNode(2)->giveCoordinate(1);
285  double y1 = this->giveNode(2)->giveCoordinate(2);
286 
287  double x2 = this->giveNode(3)->giveCoordinate(1);
288  double y2 = this->giveNode(3)->giveCoordinate(2);
289 
290  double x3 = this->giveNode(4)->giveCoordinate(1);
291  double y3 = this->giveNode(4)->giveCoordinate(2);
292 
293  double A1 = 0.5 * ((x0-x2)*(y1-y2) - (x1-x2)*(y0-y2)) ; // signed area
294  double A2 = 0.5 * ((x0-x3)*(y2-y3) - (x2-x3)*(y0-y3)) ; // signed area
295 
296  return A1 + A2 ;
297 }

299 double Quad4_U :: computeAreaAboveEnrItem(EnrichmentItem *enrItem)
300 // **************************************************************
301 // The enrItem splits Quad_U into two portions : compute the area of the
302 // portion above the enrItem.
303 // First, define the intersection points, insert into the DelaunayTree
304 // perform the tesselation => triangles => area
305 // Latest version : 2005-09-07, when solve with structured Q4 mesh.
306 {
307  GeometryEntity* myGeo = enrItem->giveMyGeo() ;
308  std::vector<Mu::Point> pts,intersect,temp ;
309 
310  // compute the intersection between element edges and enrItem
311  // -------------------------------------------------------------------------
312  for(size_t i = 0 ; i < this->giveNumberOfNodes() ; i++)
313  {
314  Mu::Point *p = this->giveNode(i+1)->makePoint() ;
315  pts.push_back(*p);
316  delete p ;
317  }
318 
319  for(size_t i = 0 ; i < pts.size() ; i++)
320  {
321  Mu::Segment s(pts[i],pts[(i+1)%pts.size()]) ;
322 
323  if(myGeo->intersects(&s))
324  {
325  temp = myGeo->intersection(&s) ;
326  intersect.insert(intersect.end(),temp.begin(),temp.end()) ;
327  }
328  }
329  // -------------------------------------------------------------------------
330 
331  // If crack touched element edge , the above code just gives one intersection !!!
332  // -------------------------------------------------------------------------
333  PiecewiseLinear *geo = static_cast<PiecewiseLinear*>(enrItem->giveMyGeo());
334 
335  if (intersect.size() == 1)
336  {
337  std::cout << " Crack tips touched element edge !!! " << std::endl ;
338  std::list<Mu::Point*> *vertices = geo->giveMyListOfVertices() ;
339  std::list<Mu::Point*>::iterator j = vertices->end();
340  Mu::Point *lastVertex = *(--j) ;
341  lastVertex->print() ; std::cout << std::endl ; // debug only
342  for(size_t k = 0 ; k < pts.size() ; k++)
343  {
344  Mu::Segment s(pts[k],pts[(k+1)%pts.size()]) ;
345  s.print() ;
346  if(s.on(lastVertex))
347  {
348  std::cout << " found edge containing tip " << endl ;
349  intersect.push_back(*lastVertex);
350  k = pts.size() ;
351  }
352  //for( std::list<Mu::Point*>::iterator j = vertices->begin() ; j != vertices->end() ; j++)
353  // if(s.on(*j))
354  // {
355  // intersect.push_back(**j);
356  // j = vertices->end() ; // stop, no test anymore.
357  // }
358  }
359  } // end of if (intersect.size() == 1)
360  // -------------------------------------------------------------------------
361 
362  // just do for the case enrItem completely splits the element => always have
363  // 2 intersection points
364  if (intersect.size() != 2)
365  {
366  std::cout << "enrItem must completely splits the element !!! " << std::endl;
367  assert(false);
368  }
369 
370  Mu::Point *p1 = this->giveNode(1)->makePoint();
371  Mu::Point *p2 = this->giveNode(2)->makePoint();
372  Mu::Point *p3 = this->giveNode(3)->makePoint();
373  Mu::Point *p4 = this->giveNode(4)->makePoint();
374  DelaunayTree *dt = new DelaunayTree(p1,p2,p3);
375  // insert points into DelaunayTree
376  dt->insert(p4) ;
377  dt->insert(&intersect[0]) ;
378  dt->insert(&intersect[1]) ;
379 
380  // looking for kink points !!! 2005-09-18
381  std::list<Mu::Point*> *vertices = geo->giveMyListOfVertices();
382  if(vertices->size() != 2)
383  {
384  for(std::list<Mu::Point*>::iterator i = vertices->begin() ; i != vertices->end(); i++)
385  {
386  if(this->isWithinMe(*i) && (isAligned(*i,&intersect[0],&intersect[1]) == false))
387  dt->insert(*i) ;
388  }
389  }
390 
391  double x1 = intersect[0].x ; // first intersection point
392  double y1 = intersect[0].y ; // first intersection point
393 
394  double x2 = intersect[1].x ; // second intersection point
395  double y2 = intersect[1].y ; // second intersection point
396 
397  vector<DelaunayTriangle *> *v = dt->getTriangles();
398  double A = 0.0 ; double delta ;
399  // loop on delaunay triangles and compute the accumulated area
400  for(size_t i = 0 ; i < v->size() ; i++)
401  {
402  if((*v)[i]->isTriangle == true)
403  {
404  Mu::Point *center = (*v)[i]->getCenter();
405  if(x2 > x1)
406  delta = (x1-center->x)*(y2-center->y)-(x2-center->x)*(y1-center->y);
407  else
408  delta = (x2-center->x)*(y1-center->y)-(x1-center->x)*(y2-center->y);
409 
410  if(delta > 0.000001)
411  A += (*v)[i]->area();
412  }
413  }
414  delete p1 ; delete p2 ; delete p3 ; delete p4 ;
415  delete dt ; delete v ; // Purify,14-10-2005
416 
417  return A ;
418
