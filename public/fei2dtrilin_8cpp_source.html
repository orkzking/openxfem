44 FloatMatrix* FEI2dTriLin :: evaldNdx(Domain* d,IntArray* Nodes,Mu::Point* lcoords)
45 // ********************************************************************************
46 // computes the derivatives of shape functions w.r.t global coordinates
47 // dNdx = [N1,x N1,y
48 // N2,x N2,y
49 // N3,x N3,y]
50 {
51  FloatMatrix *jacobianMatrix = giveJacobianMatrixAt(d,Nodes,lcoords);
52 
53  FloatMatrix *inv = jacobianMatrix->GiveInverse();
54 
55  FloatArray *dNdKsi = giveDerivativeKsi();
56  FloatArray *dNdEta = giveDerivativeEta();
57 
58  FloatMatrix * answer = new FloatMatrix(3,2);
59 
60  for (size_t i = 0 ; i < 3 ; i++)
61  {
62  answer->at(i+1,1) = (*dNdKsi)[i]*inv->at(1,1) + (*dNdEta)[i]*inv->at(1,2);
63  answer->at(i+1,2) = (*dNdKsi)[i]*inv->at(2,1) + (*dNdEta)[i]*inv->at(2,2);
64  }
65 
66  delete jacobianMatrix ;
67  delete inv ;
68  delete dNdKsi ; // Purify, 11-10-05
69  delete dNdEta ; // Purify, 11-10-05
70 
71  return answer;
72 }

100 FloatMatrix* FEI2dTriLin :: giveJacobianMatrixAt(Domain* d,IntArray* nodes,Mu::Point* lcoords)
101 // *******************************************************************************************
102 // computes the Jacobian matrix.
103 {
104  FloatMatrix *jacobianMatrix = new FloatMatrix(2,2);
105 
106  FloatArray *dNdKsi = giveDerivativeKsi();
107  FloatArray *dNdEta = giveDerivativeEta();
108 
109  for (size_t i = 0 ; i < 3 ; i++)
110  {
111  double x = d -> giveNode((*nodes)[i])->giveCoordinate(1);
112  double y = d -> giveNode((*nodes)[i])->giveCoordinate(2);
113 
114  jacobianMatrix->at(1,1) += (*dNdKsi)[i]*x;
115  jacobianMatrix->at(1,2) += (*dNdKsi)[i]*y;
116  jacobianMatrix->at(2,1) += (*dNdEta)[i]*x;
117  jacobianMatrix->at(2,2) += (*dNdEta)[i]*y;
118  }
119 
120  delete dNdKsi ; delete dNdEta ;
121 
122  return jacobianMatrix ;
123 }

148 // ******************************************************************************
149 // P : point with global(physical) coordinates x.
150 // computes its local coordinates (xi,eta).
151 // From the FE interpolation x = N_{i}(xi,eta) * x_{i}, solve this equa. with
152 // variables are xi, eta using Crame method.
153 {
154 
155  double x1 = d -> giveNode((*nodes)[0])->giveCoordinate(1);
156  double y1 = d -> giveNode((*nodes)[0])->giveCoordinate(2);
157 
158  double x2 = d -> giveNode((*nodes)[1])->giveCoordinate(1);
159  double y2 = d -> giveNode((*nodes)[1])->giveCoordinate(2);
160 
161  double x3 = d -> giveNode((*nodes)[2])->giveCoordinate(1);
162  double y3 = d -> giveNode((*nodes)[2])->giveCoordinate(2);
163 
164  double x = coords->x ;
165  double y = coords->y ;
166 
167  double A = (x2-x1)*(y3-y1)-(x3-x1)*(y2-y1) ;
168  double A1 = (x-x1) *(y3-y1)-(x3-x1)*(y-y1) ;
169  double A2 = (x2-x1)*(y-y1) -(x-x1) *(y2-y1) ;
170 
171  Mu::Point* answer = new Mu::Point(A1/A,A2/A);
172 
173  // for sure, check if the transformed point are really in the local coord. system
174  //assert(answer->x >= 0 && answer->x <= 1 && answer->y >= 0 && answer->y <= 1);
175 
176  return answer;
